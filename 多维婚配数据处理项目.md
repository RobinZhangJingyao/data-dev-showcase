### 6.哪个地区与外地婚配最多？
```sql
with table_hy as (
  select * from DWD_qzsmzj
  where length(cert_num_man) in (15, 18) and length(cert_num_woman) in (15, 18)and (op_type ="结婚登记类" or op_type="离婚申请"or op_type="补发结婚登记证类")
)

select 
       region,
       count(*)as emc
FROM(
       select 
              left(cert_num_man, 4) as region, 
              left(cert_num_woman, 4) as spouse_location
       FROM table_hy
       where left(cert_num_man, 4)<>left(cert_num_woman, 4)
  
       union ALL
  
       SELECT 
              left(cert_num_man, 4) as spouse_location, 
              left(cert_num_woman, 4) as region
       FROM table_hy
       where left(cert_num_man, 4)<>left(cert_num_woman, 4) 
    )
group by region
order by emc desc

```
### 7.统计泉州与外地具体哪个市区的婚配对数最多

 with table_hy as (
  select * from DWD_qzsmzj

  SELECT 
         sum(emc)
  from(
         select 
                spouse_location,
                count(*)as emc

                select 
                       left(cert_num_woman,4) as spouse_location
                FROM table_hy
                where left(cert_num_man,4)<>3505 and left(cert_num_woman,4)=3505 
                union all
                SELECT 
                       left(cert_num_man,4) as spouse_location
                FROM table_hy
                where left(cert_num_man,4)=3505 and left(cert_num_woman,4)<>3505 
             ) t
         group by spouse_location
         order by emc desc
      )
